# todoapp
CodingTest

Dependancies:
  Python3.8
  npm 6.4.0
  node v8.10.0
  
  
Getting Started:

#python requirements
pip install -r requirements.txt

#javascript requirements
npm install

#database migration
python3 manage.py migrate

#create super user
python3 manage.py createsuperuser

#local run
python3 manage.py runserver
npm run dev

#Public
http://localhost:8000/

#Admin
http://localhost:8000/admin/
